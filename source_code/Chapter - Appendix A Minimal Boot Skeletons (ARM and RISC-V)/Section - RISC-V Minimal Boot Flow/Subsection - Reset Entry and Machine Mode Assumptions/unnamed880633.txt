.section .text.boot
.global _start
.balign 4

_start:
    // Park all harts except hart 0
    csrr a0, mhartid
    bnez a0, secondary_park

    // Disable all interrupts
    csrw mie, zero
    csrw mip, zero

secondary_park:
    wfi
    j secondary_park
